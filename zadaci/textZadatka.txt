for i := 1 to n do 
    begin
        k := n;
        for j := 1 to i do
            begin
            k := k div 2 + n mod 2;
            end
        for j := 1 to k do 
            begin
                m := 1;
                while m < k do 
                begin
                    m := m + 1;
                end
            end;
    end;
answer:n^2
